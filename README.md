# How to play
 - Q/E = Roll
 - W = Forward
 - C = switch camera
 - pause = pause game
 
# dev tools
 - PSpeed = Forward speed
 - LRSpeed = Lock Rate Speed (Mouse sensitivity)
 - SSpeed = Strate Speed
 - HSpeed = Hover Speed
 - RSpeed = Roll Speed
 - RAcceleration = Roll Acceleration
